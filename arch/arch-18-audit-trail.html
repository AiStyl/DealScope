Event Schema

audit_events {
  id:           uuid        -- Primary key
  timestamp:    timestamptz -- Immutable creation time
  event_type:   enum        -- ai_decision, user_action, system
  actor_type:   enum        -- user, agent, system
  actor_id:     uuid        -- User ID or Agent ID
  deal_id:      uuid        -- Associated deal
  document_id:  uuid?       -- Optional document reference
  action:       text        -- What happened
  input:        jsonb       -- Input data/context
  output:       jsonb       -- Result/decision
  confidence:   float?      -- AI confidence score
  model_used:   text?       -- claude/gpt-4/gemini
  hash:         text        -- SHA-256 for integrity
  prev_hash:    text        -- Chain link to previous event
}
