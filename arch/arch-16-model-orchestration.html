Implementation Example

TypeScript lib/orchestrator/router.ts 
const ROUTING_RULES: Record<TaskType, ModelConfig> = {
  contract_analysis: { primary: 'claude', validator: 'gpt-4' },
  financial_modeling: { primary: 'gpt-4', validator: 'claude' },
  document_search:    { primary: 'gemini', validator: 'claude' },
  high_stakes:        { primary: 'all', validator: 'consensus' }
};

async function routeTask(task: Task): Promise<Result> {
  const config = ROUTING_RULES[classifyTask(task)];
  
  if (config.primary === 'all') {
    // High-stakes: run all 3 models, vote on result
    const results = await Promise.all([
      callClaude(task), callGPT4(task), callGemini(task)
    ]);
    return buildConsensus(results);
  }

  // Standard: primary model + validator
  const primary = await callModel(config.primary, task);
  const validation = await callModel(config.validator, {
    ...task, validate: primary.output
  });

  return { ...primary, confidence: validation.score };
}
