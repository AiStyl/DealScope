Implementation Example

TypeScript lib/pipeline/processDocument.ts 
async function processDocument(job: QueueJob) {
  await updateStatus(job.id, 'processing');

  // 1. Extract text from PDF
  const text = await pdfjs.extractText(job.fileUrl);

  // 2. Classify document type
  const docType = await gemini.classify(text.slice(0, 2000));

  // 3. Chunk for embedding
  const chunks = chunkText(text, { size: 500, overlap: 50 });

  // 4. Generate embeddings & store
  for (const chunk of chunks) {
    const embedding = await openai.embed(chunk);
    await supabase.from('document_chunks').insert({
      document_id: job.documentId,
      content: chunk,
      embedding
    });
  }

  await updateStatus(job.id, 'complete');
}
