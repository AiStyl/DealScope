TypeScript lib/simulation/monteCarlo.ts 
function runMonteCarloSimulation(inputs: SimInputs, iterations = 10000) {
  const results: number[] = [];

  for (let i = 0; i < iterations; i++) {
    // Sample from distributions
    const revenue = sampleNormal(inputs.revenue.mean, inputs.revenue.std);
    const multiple = sampleTriangular(inputs.multiple);
    const synergies = sampleUniform(inputs.synergies.min, inputs.synergies.max);

    // Calculate deal value
    const ev = revenue * multiple + synergies;
    results.push(ev);
  }

  return {
    p10: percentile(results, 10),
    p50: percentile(results, 50),
    p90: percentile(results, 90),
    mean: average(results)
  };
}
